function startAudit(dept){
  D.activeDept=dept;
  D.audRows=D.audRows.filter(r=>r.dept!==dept);
  document.getElementById('audit-dept-title').textContent=dept;

function makeQCard(rid,dept,qid,inv,foco,ev,om,nc,isExtra){
  const div=document.createElement('div');
  div.id='qc-'+rid;
  const isNew=inv.startsWith('[NEW]');
  const clean=inv.replace('[NEW]','').trim();
  const sc=nc==='NC'?'state-nc':om==='OM'?'state-om':(nc==='OK'||om==='OK')?'state-ok':'';
  div.className='q-card '+sc;
  div.innerHTML=`
    <div class="q-header">
      <span class="q-num">${qid||'?'}</span>
      <span class="q-text">${clean}${isNew?'<span class="q-new">NEW</span>':''}</span>
      <button class="q-del" onclick="delQCard('${rid}')" title="Remover">✕</button>
    </div>
    <div><span class="q-foco">📌 ${foco}</span></div>
    <div class="q-body">
      <div class="q-ev">
        <div class="ev-label">Constatação / Evidência</div>
        <textarea class="ev-textarea" placeholder="Descreva o que foi observado…" onchange="saveQCard('${rid}')">${ev}</textarea>
      </div>
      <div class="q-btns">
        <div class="q-btns-label">OM</div>
        <div class="q-btns-row">
          <button class="tog${om==='OM'?' t-om':''}" onclick="togQ(this,'OM','${rid}')">OM</button>
          <button class="tog${om==='OK'?' t-ok':''}" onclick="togQ(this,'OK','${rid}')">OK</button>
        </div>
        <div class="q-btns-label" style="margin-top:.5rem">NC</div>
        <div class="q-btns-row">
          <button class="tog${nc==='NC'?' t-nc':''}" onclick="togQ(this,'NC','${rid}')">NC</button>
          <button class="tog${nc==='OK'?' t-ok':''}" onclick="togQ(this,'OK','${rid}')">OK</button>
        </div>
      </div>
    </div>`;
  return div;
}

function togQ(btn,val,rid){
  const pair=btn.closest('.q-btns-row');
  const wasOn=btn.classList.contains('t-om')||btn.classList.contains('t-nc')||btn.classList.contains('t-ok');
  pair.querySelectorAll('.tog').forEach(b=>b.classList.remove('t-om','t-nc','t-ok'));
  if(!wasOn) btn.classList.add(val==='OM'?'t-om':val==='NC'?'t-nc':'t-ok');
  saveQCard(rid);
  updateQCardState(rid);
}

function updateQCardState(rid){
  const card=document.getElementById('qc-'+rid);if(!card)return;
  const rows=card.querySelectorAll('.q-btns-row');
  const omOn=rows[0]?.querySelector('.t-om');
  const ncOn=rows[1]?.querySelector('.t-nc');
  const okAny=card.querySelector('.t-ok');
  card.className='q-card';
  if(ncOn) card.classList.add('state-nc');
  else if(omOn) card.classList.add('state-om');
  else if(okAny) card.classList.add('state-ok');
}

function saveQCard(rid){
  const card=document.getElementById('qc-'+rid);if(!card)return;
  const dept=D.activeDept||'';
  const qid=card.querySelector('.q-num')?.textContent||'';
  const inv=card.querySelector('.q-text')?.textContent?.replace('NEW','').trim()||'';
  const foco=card.querySelector('.q-foco')?.textContent?.replace('📌 ','').trim()||'';
  const ev=card.querySelector('textarea')?.value||'';
  const rows=card.querySelectorAll('.q-btns-row');
  const omVal=rows[0]?.querySelector('.t-om,.t-ok')?.textContent||'';
  const ncVal=rows[1]?.querySelector('.t-nc,.t-ok')?.textContent||'';
  const audId='';
  const idx=D.audRows.findIndex(r=>r.rid===rid);
  const obj={rid,audId,dept,qid,inv,foco,ev,om:omVal,nc:ncVal};
  if(idx>=0)D.audRows[idx]=obj;else D.audRows.push(obj);
}

function delQCard(rid){

}

function addCfgQ(dept){
  const inv=prompt('Questão de Auditoria:');if(!inv?.trim())return;
  const foco=prompt('Foco / Requisito (ex: 7.5.3):','')||'';
  D.questoes[dept].push({id:`${dept}-Q${D.questoes[dept].length+1}`,inv:inv.trim(),foco});
  renderCfgRows(dept);
}
function delCfgQ(dept,i){D.questoes[dept].splice(i,1);renderCfgRows(dept);}
function addDept(){
  const d=document.getElementById('new-dept-input').value.trim();
  if(!d)return;if(D.depts.includes(d)){alert('Já existe');return;}
  D.depts.push(d);D.questoes[d]=[];
  document.getElementById('new-dept-input').value='';
  renderQuestoes();
}
function addQuestion(){
  const dept=D.cfgFilter==='Todos'?D.depts[0]:D.cfgFilter;
  if(!dept)return;
  addCfgQ(dept);
}

function exportCSV(type){
  let headers=[],rows=[];
  if(type==='intro'){
    headers=['ID','DEPARTAMENTO','AUDITOR Coordenador','AUDITOR','AUDITADO','REGISTADO POR:','HORA PREVISTA','HORA DA AUDITORIA','DATA DE AUDITORIA'];
    rows=D.introRows.map(r=>[r.id,r.dept,r.coord,r.aud,r.auditado,r.reg,r.hprev,r.hreal,r.data]);
  }else{
    headers=['ID','DEPARTAMENTO','Investigação (Questão de Auditoria)','Foco / Requisito','CONSTATAÇÃO / EVIDÊNCIA','OM','NC'];
    const dept=document.getElementById('rel-dept')?.value||'';
    const rs=dept?D.audRows.filter(r=>r.dept===dept):D.audRows;
    rows=rs.map(r=>[r.qid,r.dept,r.inv,r.foco,r.ev,r.om,r.nc]);
  }
  const csv=[headers,...rows].map(r=>r.map(c=>`"${(c||'').replace(/"/g,'""')}"`).join(',')).join('\n');
  const a=document.createElement('a');
  a.href=URL.createObjectURL(new Blob(['\ufeff'+csv],{type:'text/csv;charset=utf-8;'}));
  a.download=`audithub_${type}_${new Date().toISOString().slice(0,10)}.csv`;
  a.click();
}
